#! /bin/bash
#
# EC2 user-data — thin AWS wrapper around the shared, portable installer at
# scripts/install.sh, which is fetched at boot from raw GitHub at a pinned ref
# (the script is NOT embedded — that blew past EC2's 16 KB user-data cap). All
# the WireGuard server + dashboard provisioning logic lives in that script; this
# wrapper only handles the AWS-specific glue: IMDSv2 metadata, mapping the
# Terraform-substituted values onto the script's env contract, fetching + running
# the script as an isolated subprocess, installing the AWS CLI, and signalling
# readiness to S3 for the health check.

# Wait for background updates to finish (unattended-upgrades) before our own
# apt work below. The fetched install.sh waits again internally before its
# package installs, so this only covers the curl/unzip install in this wrapper.
echo "Waiting for apt lock..."
while fuser /var/lib/dpkg/lock-frontend >/dev/null 2>&1; do
  sleep 5
done

# curl is needed for the IMDSv2 reads just below; unzip for the AWS CLI install
# at the tail. Both are AWS-wrapper concerns, separate from install.sh's own
# package set (wireguard/wireguard-tools, and curl only on the dashboard path).
apt-get update -y
apt-get install -y curl unzip

# --- AWS instance metadata (IMDSv2) ----------------------------------------
# 1. Get the IMDSv2 Token
TOKEN=$(curl -X PUT "http://169.254.169.254/latest/api/token" -H "X-aws-ec2-metadata-token-ttl-seconds: 21600")

# 2. Use the token to get the Instance ID (consumed by the S3 .ready signal).
INSTANCE_ID=$(curl -H "X-aws-ec2-metadata-token: $TOKEN" -s http://169.254.169.254/latest/meta-data/instance-id)
echo $INSTANCE_ID > /tmp/instance-id

# --- install.sh env contract ------------------------------------------------
# One export per Terraform-substituted value, mapped onto the UPPER_CASE env
# names install.sh reads. Scalars are single tokens, double-quoted so any
# spaces / shell-special characters in the substituted value are preserved
# literally. DASHBOARD_PORT is intentionally left unset so the script applies
# its own default (8080) — matching the previous hardcoded 172.16.15.1:8080.
export WG_SERVER_NET="${wg_server_net}"
export WG_SERVER_PORT="${wg_server_port}"
export WG_SERVER_PRIVATE_KEY="${wg_server_private_key}"
export DASHBOARD_RELEASE_TAG="${dashboard_release_tag}"
export CLIENT_MANAGEMENT_MODE="${client_management_mode}"
export CLIENT_STORE_S3_BUCKET="${client_store_s3_bucket}"
export CLIENT_STORE_S3_KEY="${client_store_s3_key}"
export DASHBOARD_RELEASE_REPO="${github_repo}"
export DASHBOARD_HOST_LABEL="${dashboard_host_label}"
export DASHBOARD_ALERT_DISK_PCT="${dashboard_alert_disk_pct}"
export DASHBOARD_ALERT_CPU_PCT="${dashboard_alert_cpu_pct}"
export DASHBOARD_ALERT_CPU_SUSTAIN="${dashboard_alert_cpu_sustain}"
export DASHBOARD_ALERT_TRANSFER_BYTES="${dashboard_alert_transfer_bytes}"
export DASHBOARD_WEBHOOK_URL="${dashboard_webhook_url}"
export DASHBOARD_SLACK_BOT_TOKEN="${dashboard_slack_bot_token}"
export DASHBOARD_SLACK_CHANNEL="${dashboard_slack_channel}"
export DASHBOARD_TELEGRAM_TOKEN="${dashboard_telegram_token}"
export DASHBOARD_TELEGRAM_CHAT_ID="${dashboard_telegram_chat_id}"
export DASHBOARD_DISCORD_WEBHOOK_URL="${dashboard_discord_webhook_url}"

# Multi-line values (peer stanzas, clients JSON) are marshalled through a quoted
# heredoc so embedded newlines survive intact. Terraform's templatefile() has
# already substituted $${peers} / $${clients_json} at apply time; the quoted
# 'EOF' delimiter then stops bash from re-expanding anything in that literal
# content (no second round of $-/backtick interpretation on the substituted
# value). install.sh writes these verbatim into wg0.conf / clients.json.
WG_PEERS="$(cat <<'PEERS_EOF'
${peers}
PEERS_EOF
)"
export WG_PEERS

CLIENTS_JSON="$(cat <<'CLIENTS_EOF'
${clients_json}
CLIENTS_EOF
)"
export CLIENTS_JSON

# --- Fetch and run the shared installer as a subprocess ---------------------
# install.sh is pulled from raw GitHub at the Terraform-pinned repo + ref
# (github_repo / install_script_ref are templatefile substitutions, so a
# single-brace interpolation is correct). A download failure aborts the boot so
# a missing or wrong-ref script never reaches the run step. Running it with
# `bash <file>` (not sourced) isolates its `set -euo pipefail` from this wrapper,
# while the `|| { ...; exit 1; }` propagates a hard failure so the .ready signal
# below is never emitted on a failed install.
mkdir -p /opt/wg-install
if ! curl -fsSL "https://raw.githubusercontent.com/${github_repo}/${install_script_ref}/scripts/install.sh" -o /opt/wg-install/install.sh; then
  echo "FATAL: failed to download install.sh" >&2
  exit 1
fi
bash /opt/wg-install/install.sh || { echo "FATAL: install.sh failed" >&2; exit 1; }

# --- AWS CLI (arch-aware: AWS publishes one installer per `uname -m` value) --
case "$(uname -m)" in
  x86_64)
    AWSCLI_ARCH=x86_64
    ;;
  aarch64)
    AWSCLI_ARCH=aarch64
    ;;
  *)
    echo "FATAL: unsupported CPU architecture $(uname -m)" >&2
    exit 1
    ;;
esac
curl "https://awscli.amazonaws.com/awscli-exe-linux-$AWSCLI_ARCH.zip" -o "awscliv2.zip"
unzip -q awscliv2.zip
./aws/install

# --- Health-check signal ----------------------------------------------------
# Check WireGuard status twice with a 10s delay; on success copy a .ready marker
# to the S3 health-check bucket so the Terraform-side check knows the box is up.
for i in {1..2}; do
  if systemctl is-active --quiet wg-quick@wg0.service; then
    echo "WireGuard is active on attempt $i. Signaling S3..." > /tmp/ready
    aws s3 cp /tmp/ready s3://${health_check_bucket}/$INSTANCE_ID.ready
    exit 0
  fi

  if [ $i -lt 2 ]; then
    echo "WireGuard not active yet. Retrying in 10 seconds..."
    sleep 10
  fi
done

echo "WireGuard failed to start after 2 attempts."
exit 1
